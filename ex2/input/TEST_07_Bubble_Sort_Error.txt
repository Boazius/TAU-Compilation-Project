array IntArray = int[];

void BubbleSort(IntArray arr, int size)
{
    int i := 0;
    int min := -32767;
    int minIndex := -1;
    
    while (i < size)
    {
        int j := i+;
        while (j<size)
        {
            if (arr[j] < min)
            {
                int i := arr[minIndex];
                arr[minIndex] := arr[j];
                arr[j] := i;
            }
        }
    }
}
